import json
with open('n8n_node.json') as f:
    wf = json.load(f)
for n in wf['nodes']:
    if n.get('name') == 'Prepare Telegram Payload':
        js = n['parameters']['jsCode']
        lines = js.split('\n')
        with open('debug_output.txt', 'w') as out:
            for i in range(min(20, len(lines))):
                out.write(f'{i}: {repr(lines[i])}\n')
